DOCUMENTATION

SetUp

Require php file:

require("untappd.class.php");

example

<?php require("untappd.class.php); ?>

GetUser (username, preference, compact)

username: (string, *required)

preference: (string, *optional) info, badged, beers, checkins, friends

compact: (string, *optional) only if passing 'info' as seccond parameter true, false

returns:array

Get user by id

Untappd::GetUser("USERNAME");

Get friends from user

Untappd::GetUser("USERNAME", "friends");

Get badges from user

Untappd::GetUser("USERNAME", "badges");

Get beers from user

Untappd::GetUser("USERNAME", "beers");

Get checkins from user

Untappd::GetUser("USERNAME", "checkins");

example

<?php 
 $user = Untappd:GetUser("ravaga"); 
 echo ($user);
 ?>

GetBrewery (brewery_id, preference)

brewery_id: (string, *required)

preference: (string, *optional)info, checkins

compact: (string, *optional) only if passing 'info' as seccond parameter true, false

returns:array

Get brewery by id

Untappd::GetBrewery("BREWERY_ID");

Get checkins from brewery

Untappd::GetBrewery("BREWERY_ID", "checkins");

example

<?php 
 $brewery = Untappd:GetBrewery("1122"); 
 echo ($brewery);
 ?>

BrewerySearch (keyword, limit, offset)

keyword: (string, *required)

limit: (int, *optional)Min 1 ~ Max 50 (default 10)

offset: (int, *optional)

returns:array

Search brewery

Untappd::BrewerySearch("KEYWORD");

Advance Search for brewery

Untappd::BrewerySearch("KEYWORD", 12, 4);

example

<?php 
 $brewery = Untappd:BrewerySearch("stone brewery"); 
 echo ($brewery);
 ?>

BeerSearch (keyword, limit, offset)

keyword: (string, *required)

limit: (int, *optional)Min 1 ~ Max 50 (default 10)

offset: (int, *optional)

returns:array

Search for beer

Untappd::BeerSearch("KEYWORD");

Advance Search for beer

Untappd::BeerSearch("KEYWORD", 12, 4);

example

<?php 
 $beer = Untappd:BeerSearch("lagunitas");

GetVenue (venue_id, compact)

venue_id: (int, *required)

compact: (string, *optional)true, false

returns:array

Get venue by id

Untappd::GetVenue("VENUE_ID");

Get all info from venue

Untappd::GetVenue("VENUE_ID", "false");

example

<?php
